# virtual_pet.py # Step 0: Set up the pet pet_name = input("Give your pet a name: ") hunger = 5 # 0 = full, 10 = starving happiness = 5 # 0 = sad, 10 = super happy energy = 5 # 0 = exhausted, 10 = full energy # Step 1: Main loop while True: print("\n---", pet_name, "Status ---") print(f"Hunger: {hunger}/10") print(f"Happiness: {happiness}/10") print(f"Energy: {energy}/10") print("\nWhat do you want to do?") print("1. Feed") print("2. Play") print("3. Sleep") print("4. Quit") choice = input("Choose 1-4: ") if choice == "1": print(f"You feed {pet_name}. Yum!") hunger = max(0, hunger - 2) # Feeding reduces hunger happiness = min(10, happiness + 1) elif choice == "2": print(f"You play with {pet_name}. Fun!") happiness = min(10, happiness + 2) energy = max(0, energy - 2) hunger = min(10, hunger + 1) elif choice == "3": print(f"{pet_name} takes a nap. Zzz...") energy = min(10, energy + 3) hunger = min(10, hunger + 1) elif choice == "4": print(f"Goodbye! {pet_name} will miss you!") break else: print("Invalid choice. Try again!") # Optional: Check if pet is unhappy or starving if hunger >= 10: print(f"{pet_name} is starving! Feed it soon!") if energy <= 0: print(f"{pet_name} is exhausted! Let it sleep!") if happiness <= 0: print(f"{pet_name} is very sad... Play with it!")
